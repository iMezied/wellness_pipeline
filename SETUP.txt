# setup.sh — سكريبت الإعداد الأول على الـ VPS
# ============================================================
#!/bin/bash
set -e

echo "📦 تثبيت المتطلبات..."
apt-get update -qq
apt-get install -y ffmpeg python3-pip mysql-server redis-server

echo "🐍 تثبيت Python packages..."
pip3 install -r requirements.txt

echo "📁 إنشاء المجلدات..."
mkdir -p /var/wellness/output /var/wellness/tmp
chown -R www-data:www-data /var/wellness

echo "🗄️ إعداد قاعدة البيانات..."
mysql -u root -e "CREATE DATABASE IF NOT EXISTS wellness_pipeline CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;"
mysql -u root -e "CREATE USER IF NOT EXISTS 'wellness_user'@'localhost' IDENTIFIED BY '${DB_PASS}';"
mysql -u root -e "GRANT ALL ON wellness_pipeline.* TO 'wellness_user'@'localhost';"

echo "🚀 تهيئة قاعدة البيانات..."
python3 -c "from db import init_db; init_db()"

echo "✅ الإعداد اكتمل!"
echo ""
echo "للاختبار:    python3 pipeline.py once"
echo "للإنتاج:     python3 pipeline.py"


# ============================================================
# /etc/systemd/system/wellness-pipeline.service
# ضع هذا الملف على الـ VPS لتشغيل تلقائي عند الإقلاع
# ============================================================
[Unit]
Description=Wellness Content Pipeline
After=network.target mysql.service redis.service

[Service]
Type=simple
User=www-data
WorkingDirectory=/var/wellness/pipeline
EnvironmentFile=/var/wellness/pipeline/.env
ExecStart=/usr/bin/python3 pipeline.py
Restart=always
RestartSec=30
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target

# تفعيل الخدمة:
